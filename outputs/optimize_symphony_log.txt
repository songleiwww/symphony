**实现方案**：统一错误处理中间件

```python
# error_handler.py
class SymphonyErrorHandler:
    def __init__(self, openclaw_message_bus):
        self.message_bus = openclaw_message_bus
    
    def handle_error(self, error, context=None):
        error_report = {
            "tool": "symphony",
            "timestamp": datetime.now().isoformat(),
            "error_type": type(error).__name__,
            "message": str(error),
            "context": context,
            "suggestions": self._generate_suggestions(error)
        }
        
        # 发送到OpenClaw消息系统
        self.message_bus.publish("tool_error", error_report)
        
        # 存储到memory系统
        openclaw_memory.store("tool_errors", error_report)
        
        return self._format_user_friendly_error(error)
```

## 4. 消息与memory系统集成

**优化方案**：双向通信机制

```python
# symphony_integration.py
class SymphonyIntegration:
    def __init__(self):
        self.message_bus = OpenClawMessageBus()
        self.memory = OpenClawMemory()
        
    def send_analysis_result(self, result):
        # 结构化结果存入memory
        memory_key = f"symphony_analysis_{datetime.now().timestamp()}"
        self.memory.store(memory_key, {
            "result": result,
            "tool": "symphony",
            "timestamp": datetime.now().isoformat()
        })
        
        # 发送实时通知
        self.message_bus.publish("tool_completed", {
            "tool": "symphony",
            "status": "success",
            "summary": self._generate_summary(result),
            "memory_reference": memory_key
        })
    
    def get_historical_context(self):
        # 从memory获取相关历史记录
        return self.memory.query({
            "tool": "symphony",
            "limit": 5,
            "sort_by": "timestamp_desc"
        })
```

```json
  "input_schema": {
    "type": "object",
    "properties": {
      "prompt": { "type": "string", "description": "音乐风格或内容的提示词" },
      "session_id": { "type": "string", "description": "用于集成OpenClaw Memory的会话ID" }

    "required": ["prompt"]
```

```python
def handle_request(args):
    try:
        session_id = args.get("session_id")
        # 核心逻辑
        audio_url = generate_music(args["prompt"])
        
        # 集成Memory：将结果写入上下文
        if session_id:
            openclaw.memory.update(session_id, {"last_audio": audio_url})

    except Exception as e:
        # 统一错误反馈
```
